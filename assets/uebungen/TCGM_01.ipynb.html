def sineToneGen(freq,amp,length,freqmult,sr):
  samp_length = int(length * sr)
  arr = np.zeros(samp_length)
  for i in range(samp_length):
    arr[i] = np.sin(2.0 * np.pi * float(i) / float(sr) * freq * freqmult) * amp
  return arr
