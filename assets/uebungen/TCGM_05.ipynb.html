Aufgabe 3:¶

Importiert ein Audiofile mit librosa. Teilt es in x Segmente (Teile). Randomisiert die Reihenfolge der Teile und Fadet jedes Teil ein und aus.

In [ ]:
num_segs = 1000
samp, sr = lr.load("/content/samp.wav")
total_length = len(samp)
seg_length = total_length // num_segs
random_choises = np.random.choice(num_segs, num_segs, replace=False)
segments = np.arange(0, seg_length * num_segs + 1 , seg_length)
samp_final = np.zeros(len(samp))
env = np.zeros((seg_length))
decrement = np.arange(3,-1,-1)
for i in range(seg_length):
  env[i] = np.sin(np.pi *  (i / seg_length))

index = 0
for i in random_choises:
  print(i)
  samp_final[index * seg_length : (index + 1) * seg_length] = samp[segments[i] : segments[i + 1]] * env
  index += 1

display(Audio(samp_final, rate = sr))
